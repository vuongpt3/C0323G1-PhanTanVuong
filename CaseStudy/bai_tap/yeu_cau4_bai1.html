function normalizeString(str) { // Chuyển hết thành chữ thường và tách các từ ra thành một mảng const words = str.toLowerCase().trim().split(/\s+/); // Chuyển đổi từng từ thành chuỗi chuẩn hóa const normalizedWords = words.map(word => { // Chuyển đổi ký tự đầu tiên thành chữ hoa và còn lại thành chữ thường return word.charAt(0).toUpperCase() + word.slice(1); }); // Ghép các từ lại thành một xâu return normalizedWords.join(' '); } function findLongestWord(str) { // Tách các từ ra thành một mảng const words = str.trim().split(/\s+/); // Tìm từ dài nhất trong mảng let longestWord = ''; words.forEach(word => { if (word.length > longestWord.length) { longestWord = word; } }); return longestWord; } // Hàm main function main() { const inputString = prompt('Nhập xâu ký tự:'); // Chuẩn hóa xâu const normalizedString = normalizeString(inputString); console.log(`Xâu sau khi chuẩn hóa: ${normalizedString}`); // Tìm từ dài nhất const longestWord = findLongestWord(inputString); console.log(`Từ dài nhất trong xâu: ${longestWord}`); } main();
